# Модуль містить функцію для обчислення добутку непарних чисел

def calc_y(n):
    """
    Обчислює вираз y = 1 * 3 * 5 * ... * (2n - 1)
    Параметр:
        n — натуральне число
    Повертає:
        добуток непарних чисел до (2n - 1)
    """
    y = 1
    for i in range(1, n + 1):
        y *= 2 * i - 1
    return y
